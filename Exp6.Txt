#include <stdio.h>

int main() {
    int n, i, p[10], bt[10];

    printf("Enter number of processes: ");
    scanf("%d", &n);

    for(i = 0; i < n; i++) {
        printf("Priority and Burst Time of P%d: ", i + 1);
        scanf("%d%d", &p[i], &bt[i]);
    }

    printf("\nProcess\tPriority\tBurst Time\n");
    for(i = 0; i < n; i++)
        printf("P%d\t%d\t\t%d\n", i + 1, p[i], bt[i]);

    return 0;
}
