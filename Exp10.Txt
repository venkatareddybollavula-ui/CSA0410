#include <stdio.h>
#include <sys/ipc.h>
#include <sys/msg.h>

struct message {
    long type;
    char text[50];
};

int main() {
    struct message m;

    int id = msgget(1234, 0666 | IPC_CREAT);

    m.type = 1;

    printf("Enter message: ");
    scanf("%s", m.text);

    msgsnd(id, &m, sizeof(m.text), 0);
    msgrcv(id, &m, sizeof(m.text), 1, 0);

    printf("Received Message: %s", m.text);

    return 0;
}
