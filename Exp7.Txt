#include <stdio.h>

int main() {
    int n, bt[10], i, j, temp;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    for(i = 0; i < n; i++) {
        printf("Burst Time of P%d: ", i + 1);
        scanf("%d", &bt[i]);
    }

    for(i = 0; i < n - 1; i++) {
        for(j = i + 1; j < n; j++) {
            if(bt[i] > bt[j]) {
                temp = bt[i];
                bt[i] = bt[j];
                bt[j] = temp;
            }
        }
    }

    printf("\nExecution Order:\n");
    for(i = 0; i < n; i++)
        printf("%d ", bt[i]);

    return 0;
}
